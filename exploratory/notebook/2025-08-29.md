# 2025-08-29
## Demultiplexing 16S fly data
```
 qiime tools import \
   --type EMPPairedEndSequences \
   --input-path sequences \
   --output-path multiplexed_data.qza

 qiime demux emp-paired \
   --i-seqs multiplexed_data.qza \
   --m-barcodes-file demux.tsv \
   --m-barcodes-column BarcodeSequence \
   --p-rev-comp-mapping-barcodes \
   --p-rev-comp-barcodes \
   --o-per-sample-sequences demux.qza \
   --o-error-correction-details demux_details.qza
   
qiime demux summarize \
  --i-data demux.qza \
  --o-visualization demux.qzv
```

## Denoising
```
qiime dada2 denoise-paired \
  --i-demultiplexed-seqs demux.qza \
  --p-trim-left-f 0 \
  --p-trim-left-r 0 \
  --p-trunc-len-f 220 \
  --p-trunc-len-r 180 \
  --o-table table.qza \
  --o-representative-sequences rep-seqs.qza \
  --o-denoising-stats denoising-stats.qza
```

```
qiime feature-table summarize \
  --i-table table.qza \
  --o-visualization table.qzv \
  --m-sample-metadata-file metadata.tsv

qiime feature-table tabulate-seqs \
  --i-data rep-seqs.qza \
  --o-visualization rep-seqs.qzv

qiime metadata tabulate \
  --m-input-file denoising-stats.qza \
  --o-visualization denoising-stats.qzv
```

## Taxonomy
```
qiime feature-classifier classify-sklearn \
  --i-classifier greengenes2-2024.09-515-806-nb-classifier.qza \
  --i-reads rep-seqs.qza \
  --o-classification taxonomy.qza

qiime metadata tabulate \
  --m-input-file taxonomy.qza \
  --o-visualization taxonomy.qzv
```

## Filtering
```
qiime taxa filter-table \
  --i-table table.qza \
  --i-taxonomy taxonomy.qza \
  --p-exclude mitochondria,chloroplast,sp004296775 \
  --o-filtered-table table-nmnc.qza

qiime taxa filter-table \
  --i-table table-nmnc.qza \
  --i-taxonomy taxonomy.qza \
  --p-exclude archaea \
  --o-filtered-table table-nmncna.qza

qiime taxa filter-table \
    --i-table table-nmncna.qza \
    --i-taxonomy taxonomy.qza \
    --p-exclude "Unassigned" \
    --o-filtered-table table-filtered.qza

qiime feature-table filter-features \
    --i-table table-filtered.qza \
    --p-min-frequency 10 \
    --p-min-samples 2 \
    --o-filtered-table table-filtered-all.qza
```

## Taxonomy visualization
```
qiime taxa barplot \

  --i-taxonomy taxonomy.qza \
  --m-metadata-file metadata.tsv \
  --o-visualization taxa-bar-plots.qzv
```
